```typescript
#!/usr/bin/env node
/**
 * One-shot script: registers 3 test agents on Base Sepolia AgentPrivacyRegistry.
 * Run once: npx tsx packages/crypto/scripts/register-test-agents.ts
 *
 * Env vars required:
 *   BASE_SEPOLIA_RPC — RPC endpoint (e.g. https://sepolia.base.org)
 *   PRIVATE_KEY     — Funded account private key (for gas)
 *
 * Output: packages/crypto/test-agents/agent-{1,2,3}.json
 * WARNING: Do NOT run in CI — this costs gas and agents stay registered permanently.
 */
import { prepareOnboarding, registerOnChain } from '../src/onboarding.js'
import { ethers } from 'ethers'
import { writeFileSync, mkdirSync } from 'fs'
import { join, dirname } from 'path'
import { fileURLToPath } from 'url'

const __dirname = dirname(fileURLToPath(import.meta.url))
const REGISTRY = '0x857E1049A5eE2cCA03a5C95F32089FECe51Ce8ff'
const AGENT_IDS = [1n, 2n, 3n]            // bidder, competitor, observer
const CAPABILITY_IDS = [[1n], [1n], [1n]] // single capability each

// BigInt-safe JSON serializer
function serializeState(state: unknown): string {
  return JSON.stringify(state, (_, v) => typeof v === 'bigint' ? v.toString() + 'n' : v, 2)
}

async function main() {
  const rpc = process.env.BASE_SEPOLIA_RPC
  const key = process.env.PRIVATE_KEY
  if (!rpc || !key) throw new Error('BASE_SEPOLIA_RPC and PRIVATE_KEY env vars required')

  const provider = new ethers.JsonRpcProvider(rpc)
  const signer = new ethers.Wallet(key, provider)
  const network = await provider.getNetwork()
  console.log(`Connected to chainId ${network.chainId}`)
  if (network.chainId !== 84532n) throw new Error('Expected Base Sepolia (84532)')

  const outDir = join(__dirname, '..', 'test-agents')
  mkdirSync(outDir, { recursive: true })

  for (let i = 0; i < AGENT_IDS.length; i++) {
    const agentId = AGENT_IDS[i]
    const capabilityIds = CAPABILITY_IDS[i]
    console.log(`\nPreparing agent ${agentId}...`)

    const state = await prepareOnboarding(agentId, capabilityIds)
    console.log(`  Commitment computed. Registering on-chain...`)

    try {
      await registerOnChain(state, REGISTRY, signer)
      console.log(`  Agent ${agentId} registered.`)
    } catch (err: unknown) {
      const msg = err instanceof Error ? err.message : String(err)
      if (msg.includes('AlreadyRegistered') || msg.includes('already')) {
        console.log(`  Agent ${agentId} already registered — skipping.`)
      } else {
        throw err
      }
    }

    const outPath = join(outDir, `agent-${agentId}.json`)
    writeFileSync(outPath, serializeState(state), 'utf8')
    console.log(`  State saved to ${outPath}`)
  }

  console.log('\nDone. All 3 agents registered (or already registered).')
}

main().catch(err => { console.error(err); process.exit(1) })
```
