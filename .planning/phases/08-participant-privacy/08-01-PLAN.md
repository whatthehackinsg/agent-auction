```typescript
function maskEventsForParticipant(events: Array<Record<string, unknown>>): Array<Record<string, unknown>> {
  return events.map(e => {
    // System events (agent_id='0') pass through unmodified
    if (e.agent_id === '0') return e
    // Replace agent_id with zkNullifier, remove wallet
    const { agent_id, wallet, ...rest } = e

      ...rest,

      // Omit wallet entirely (not in spread)
    }
  })
}
```
